this-Problematic

Worauf this zeigt hängt davon ab, wie die Funktion aufgerufen wird!

// Aufruf als Methode eines Objekts
var objekt = {
  prop: 37,

  foo: function() {
    return this.prop;
  }
};

console.log(objekt.foo()); // 37

this-Problematik

function Foo(){
  this.a = 37;
  return {a:38};
}

// Aufruf als Konstruktor
// this hängt am Objekt, das konstruiert wurde
ojekt = new Foo();

console.log(objekt.a); // 38

Prototypes

  • "Inheritance" in JavaScript
  • Accessing a non-existing property results in a lookup on the prototype
  • Funktions, Arrays, etc. have the prototype "Object"

Prototypes

var auto = {
  wheels: 4,
  tires: true
}

var Motorcycle = {
  this.wheels = 2;
}
Motorcycle.prototype = auto;
var ducati = new Motorcycle();

console.log(ducati.wheels); // 2

delete ducati.wheels
console.log(ducati.wheels); // 4
